    $\bullet \left(\begin{matrix} x \\ y \end{matrix}\right) \perp \left(\begin{matrix} -y\\ x \end{matrix} \right) \qquad \bullet \angle(\vec{v}, \vec{w}) = \arccos(\frac{\langle \vec{v}, \vec{w}}{\left|\vec{v}\right| \cdot \left|\vec{w}\right|})$
\end{itemize}
