    \begin{flalign}
        &\tan{\varphi} = \frac{\sin{\varphi}}{\cos{\varphi}}&\\
        &\cot{\varphi} = \frac{\cos{\varphi}}{\sin{\varphi}}&
    \end{flalign}
\end{minipage}

\begin{minipage}{0.4\linewidth}
    \begin{flalign}
        &\sinh{x} = \frac{e^{x} − e^{−x}}{2}&\\
        &\cosh{x} = \frac{e^{x} + e^{−x}}{2}&\\
        &\tanh{x} = \frac{\sinh{x}}{\cosh{x}}&\\
        &\coth{x} = \frac{\cosh{x}}{\sinh{x}}&
    \end{flalign}
\end{minipage}
