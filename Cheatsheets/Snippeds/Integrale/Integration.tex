\textbf{Lineare Integration}\\
\begin{flalign}
    &\int{f(m \cdot x + q) \,dx = \frac1m F(m \cdot x + q) + c}&\\
    &\int{(m \cdot x + q)^p \,dx = \frac{1}{m \cdot (p + 1)} \cdot (m \cdot x + q)^{p + 1} + c}&\\
    &\int{a^{m \cdot x + q} \,dx = \frac{1}{m \cdot (p + 1)} \cdot (m \cdot x + q)^{p + 1} + c}&\\
    &\int{y_0 \cdot a^{\frac{x - x_0}{\sum}}} \,dx = \frac{\sum}{\ln{a}} \cdot y_0 \cdot a^{\frac{x - x_0}{\sum}} + c&\\
    &\int{A \cdot \sin(\omega \cdot t + \varphi)} \,dt = - \frac{A}{\omega} \cdot \cos(\omega \cdot + \varphi) + c&
\end{flalign}
