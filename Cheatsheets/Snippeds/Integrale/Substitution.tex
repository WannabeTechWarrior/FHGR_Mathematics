\subsubsection{Substitution}
\textbf{Normale Substitution}\\
\begin{align}
    \int_{a}^{b} f(x) dx \qquad \large|& \qquad u(x) = f(x)\notag\\
    |& \qquad u'(x) = f'(x)\notag\\
    |& \qquad du = u'(x)\cdot dx\notag\\
    \int_{u(a)}^{u(b)} u(x) \frac{1}{u'(x)} du \qquad \large|& \qquad dx = \frac{1}{u'(x)} \cdot du
\end{align}\\
